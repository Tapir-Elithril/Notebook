    \begin{document}
\begin{frame}
\title{标题}
\author{作者}
\institute{学院}
\date{\today}
\titlepage
\end{frame}

\section{第一章}
\begin{frame}{第一节}
\begin{thm}[名称]
\hspace{2em}定理.
\end{thm}
\pause
\setbeamercolor{block title}{fg = white,bg = uibgreen}
\begin{proof}
\hspace{2em}证明.
\end{proof}
\pause
\setbeamercolor{block title}{fg = white,bg = uibblue}
\begin{block}{自定义}
\hspace{2em}自定义内容.
\end{block}
\end{frame}
\end{document}
